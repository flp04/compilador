programa
decimal notaA1, notaA2, notaA3, soma_notas, media;
escreva("PROGRAMA PARA CALCULAR A MÉDIA DO SEMESTRE");
escreva("Digite a nota da A1");
leia(notaA1);
escreva("Digite a nota da A2");
leia(notaA2);
escreva("Digite a nota da A3");
leia(notaA3);
soma_notas := (notaA1 + notaA2 + notaA3);
media := soma_notas / 3.0;
se (media >= 7.0) {
  escreva("Média: " + media);
  escreva("Aluno aprovado.");
} senao {
  escreva("Aluno reprovado.");
  escreva(media);
}
